# Simple string replacements for search preprocessor
# Format:
#   kind:key => expansion
# - kind is one of: site, is, has
# - key is a token without spaces (comma-separated in query supported for site only)
# - expansion is plain text; use parentheses and OR as needed
# Examples:
# site:yt => (youtube.com OR youtu.be OR m.youtube.com OR www.youtube.com OR youtube-nocookie.com)
# is:image => (.png OR .jpg OR .jpeg OR .gif OR .gifs OR .apng OR .webp OR .avif OR .svg)
# has:video => (.mp4 OR .webm OR .ogg OR .ogv OR .mov OR .m4v)
